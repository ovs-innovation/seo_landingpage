function sparkline(points){
  const w=240,h=64;
  const max=Math.max(...points), min=Math.min(...points);
  const step=w/(points.length-1);
  const d=points.map((p,i)=>{const x=i*step;const y=h-((p-min)/(max-min))*h;return `${i===0?\"M\":\"L\"} ${x} ${y}`;}).join(\" \");
  const area=`${d} L ${w} ${h} L 0 ${h} Z`;
  return `<svg viewBox=\"0 0 ${w} ${h}\" style=\"width:100%;height:64px\"><defs><linearGradient id=\"sp${Math.random().toString(36).slice(2,7)}\" x1=\"0\" y1=\"0\" x2=\"0\" y2=\"1\"><stop offset=\"0%\" stop-color=\"#ff4f93\" stop-opacity=\".35\"/><stop offset=\"100%\" stop-color=\"#ff4f93\" stop-opacity=\"0\"/></linearGradient></defs><path d=\"${area}\" fill=\"#ffdce8\" opacity=\".7\"/><path d=\"${d}\" fill=\"none\" stroke=\"#ff4f93\" stroke-width=\"2.2\" stroke-linecap=\"round\" stroke-linejoin=\"round\"/></svg>`;
}

/* ============ FORM SUBMIT ============
   Both forms send via mailto: by default (works anywhere, no backend).
   To use Formspree or your own API, set FORM_ENDPOINT below to your URL
   (e.g. \"https://formspree.io/f/XXXXXXX\") â€” the script will POST JSON to it. */
const FORM_ENDPOINT=\"\"; // leave empty for mailto fallback
const TARGET_EMAIL=\"hello@vastoratech.com\";

async function submitLead(data,onDone){
  if(FORM_ENDPOINT){
    try{
      const res=await fetch(FORM_ENDPOINT,{
        method:\"POST\",
        headers:{\"Content-Type\":\"application/json\",\"Accept\":\"application/json\"},
        body:JSON.stringify(data)
      });
      if(!res.ok) throw new Error(\"Bad response\");
      onDone(true);
    }catch(err){onDone(false)}
  }else{
    // mailto fallback
    const subject=encodeURIComponent(\"New SEO Audit Request â€” \"+(data.name||\"\"));
    const body=encodeURIComponent(
      `Name: ${data.name}\nEmail: ${data.email}\nWebsite: ${data.website||\"â€”\"}\nPhone: ${data.phone||\"â€”\"}\n\nMessage:\n${data.message||\"â€”\"}`
    );
    window.location.href=`mailto:${TARGET_EMAIL}?subject=${subject}&body=${body}`;
    onDone(true);
  }
}

function validate(form){
  const data={};
  let ok=true;
  form.querySelectorAll(\"input,textarea\").forEach(i=>{data[i.name]=i.value.trim()});
  if(!data.name||!data.email){ok=false}
  if(data.email && !/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(data.email)){ok=false}
  return {ok,data};
}
